# Steps to push to deployment

  - git clone

  - cd chandrikadeb7.github.io

  - source ~/.nvm/nvm.sh

  - nvm install node (version 16)

  - npm install -g yarn

  - yarn

  - npm install -g gatsby-cli

  - npm install --force

  - git add .

  - git commit -am "changes"

  - gatsby build

  - npm run deploy or npm start
